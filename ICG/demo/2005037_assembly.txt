.MODEL SMALL
.STACK 1000H
.Data
	number DB "00000$"
	w DW 10 DUP (0000H)
.CODE
main PROC
	MOV AX,@DATA
	MOV DS,AX
	PUSH BP
	MOV BP,SP
L2:
	SUB SP,2
L3:
	SUB SP,10
L4:
	MOV AX,0
	PUSH AX
	MOV AX,5
	PUSH AX
	MOV BX,2
	POP AX
	MUL BX
	MOV BX,12
	SUB BX,AX
	MOV SI,BX
	NEG SI
	POP AX
	MOV [BP+SI],AX
L5:
	MOV AX,5
	PUSH AX
	MOV AX,4
	PUSH AX
	MOV BX,2
	POP AX
	MUL BX
	MOV BX,12
	SUB BX,AX
	MOV SI,BX
	NEG SI
	POP AX
	MOV [BP+SI],AX
	MOV AX,5
	PUSH AX
	MOV BX,2
	POP AX
	MUL BX
	MOV BX,12
	SUB BX,AX
	MOV SI,BX
	NEG SI
	MOV AX,[BP+SI]
	CMP AX,1
	JGE L6
	jmp L7
L6:
L9:
	MOV AX,2
	PUSH AX
	MOV AX,5
	PUSH AX
	MOV BX,2
	POP AX
	MUL BX
	MOV BX,12
	SUB BX,AX
	MOV SI,BX
	NEG SI
	PUSH [BP+SI]
	POP BX
	POP AX
	MUL BX
	PUSH AX
	POP AX
	MOV [BP-2],AX
	jmp L8
L7:
L10:
	MOV AX,3
	PUSH AX
	MOV AX,4
	PUSH AX
	MOV BX,2
	POP AX
	MUL BX
	MOV BX,12
	SUB BX,AX
	MOV SI,BX
	NEG SI
	PUSH [BP+SI]
	POP BX
	POP AX
	MUL BX
	PUSH AX
	POP AX
	MOV [BP-2],AX
L8:
L11:
	MOV AX,[BP-2]
	CALL print_output
	CALL new_line
L1:
	ADD SP,12
	POP BP
	MOV AX,4CH
	INT 21H
main ENDP


;------print library-------;

new_line proc
    push ax
    push dx
    mov ah,2
    mov dl,0Dh
    int 21h
    mov ah,2
    mov dl,0Ah
    int 21h
    pop dx
    pop ax
    ret
    new_line endp
print_output proc  ;print what is in ax
    push ax
    push bx
    push cx
    push dx
    push si
    lea si,number
    mov bx,10
    add si,4
    cmp ax,0
    jnge negate
    print:
    xor dx,dx
    div bx
    mov [si],dl
    add [si],'0'
    dec si
    cmp ax,0
    jne print
    inc si
    lea dx,si
    mov ah,9
    int 21h
    pop si
    pop dx
    pop cx
    pop bx
    pop ax
    ret
    negate:
    push ax
    mov ah,2
    mov dl,'-'
    int 21h
    pop ax
    neg ax
    jmp print
    print_output endp
END main
