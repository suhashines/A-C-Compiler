.MODEL SMALL
.STACK 1000H
.Data
	number DB "00000$"
	a DW 1 DUP (0000H)
	b DW 1 DUP (0000H)
	c DW 1 DUP (0000H)
.CODE
func_a PROC
	PUSH BP
	MOV BP,SP
L1:
	MOV AX,1
	MOV [BP+4],AX
L2:
	MOV AX,5
	MOV [BP+6],AX
L3:
	MOV AX,3
	MOV [BP+8],AX
L4:
	SUB SP,2
L5:
	MOV AX,7
	MOV a,AX
main PROC
	MOV AX,@DATA
	MOV DS,AX
	PUSH BP
	MOV BP,SP
	CALL func_a
L6:
	ADD SP,0
	POP BP
	MOV AX,4CH
	INT 21H
main ENDP
END main
